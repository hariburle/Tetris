@echo off
REM This batch file automates the deployment process for the Tetris game.
REM To use it, rename this file from "deploy.txt" to "deploy.bat" and then double-click it.

echo Starting deployment...

REM The "npm run deploy" command is defined in package.json.
REM It first runs "npm run build" (predeploy script) to create the production files in the 'dist' folder.
REM Then, it uses the 'gh-pages' package to push the contents of the 'dist' folder to the 'gh-pages' branch of your GitHub repository.
npm run deploy

echo.
echo Deployment process finished.
echo Press any key to close this window...
pause > nul
